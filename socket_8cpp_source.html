106  bool Socket::getSocketAddress(const std::string& _url, unsigned _port, Protocol _protocol) {
107  // ------ Get proper address info -------
108  struct addrinfo addrHints;
109  mAddress = nullptr;
110  memset(&addrHints, 0, sizeof(addrHints));
111  addrHints.ai_family = AF_UNSPEC; // Connect to either ip v4 or v6
112  assert(_protocol == Protocol::TCP || _protocol == Protocol::UDP); // Only TCP is currently supported
113  switch(_protocol) {
114  case Protocol::TCP:
115  addrHints.ai_socktype = SOCK_STREAM; // Connection type TCP IP
116  addrHints.ai_protocol = IPPROTO_TCP;
117  break;
118  case Protocol::UDP:
119  addrHints.ai_socktype = SOCK_DGRAM;
120  addrHints.ai_protocol = IPPROTO_UDP;
121  break;
122  }
123  // --- Query the OS for an address fitting the description
124  std::stringstream portStream;
125  portStream << _port;
126  int res = getaddrinfo(_url.c_str(), portStream.str().c_str(), &addrHints, &mSystemAddress);
127  if(0 != res)
128  {
129  std::cout << "Error: unable to get addr info for socket\n\t-url: "
130  << _url << "\n\t-port: " << portStream.str() << "\n";
131  return false;
132  }
133 
134  return true;
135  }
